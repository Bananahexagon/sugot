# データ型

```
// 数値型
// 符号あり整数型
let a: i8  // -128 ~ 127
let b: i16 // -32768 ~ 32767
let c: i32 // -2147483648 ~ 2147483647、デフォルトの整数型
let d: i64 // -9223372036854775808 ~ 9223372036854775807
let e: int // 環境によって変動

// 符号なし整数型
let f: u8   // 0 ~ 255
let g: u16  // 0 ~ 65535
let h: u32  // 0 ~ 4294967295
let i: u64  // 0 ~ 18446744073709551615
let j: uint // 環境によって変動

// 浮動小数点型
let k: f32 // IEEE-754の単精度浮動小数点数
let l: f64 // IEEE-754の倍精度浮動小数点数

// 論理値型
let m: bool // trueかfalseのみ

// 文字列型
let n: str // UTF-8の文字列、内部的にはu8のベクタとして扱われる

// 配列型
let o: [i32; 5] // 長さ5のi32型の固定長配列

// タプル型
let p: (i32, bool, str) // (i32, bool, str)の形のタプル

// ベクタ型
let q: Vec<i32> // i32型の可変長配列

// void型
// void型の変数を宣言することはできない
let r: void // コンパイルエラー
```
